\section{Données}

\begin{frame}{Données utilisées}
\begin{itemize}
	\item \textbf{25 transcrits} dans 3 voies oncologiques : KRAS, suppresseurs tumoraux, MAPK/PI3K.
	\item \textbf{2\,516 miRNAs} matures (miRBase).
	\item $\Rightarrow$ \textbf{62\,900 paires} transcrit $\times$ miRNA à tester.
\end{itemize}

\vspace{0.5em}
\begin{block}{Résultat}
	\begin{itemize}
		\item 62\,368 paires ont $\geq$ 1 interaction $\Rightarrow$ \textbf{1\,208\,803 interactions}.
		\item Dont 809\,828 avec seed canonique (67\,\%) et 398\,975 \alert{seedless} (33\,\%).
		\item Les interactions seedless sont exclues du clustering.
	\end{itemize}
\end{block}
\end{frame}
