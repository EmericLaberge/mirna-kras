\begin{frame}{NMF — Sélection du rang}
\begin{itemize}
  \item Rang testé : $k = 2, 3, \dots, 25$ (maximum = 25 features)
  \item Critère : corrélation cophénétique
  \item \textbf{Rang optimal} : $k = 2$ (cophénétique = 0.9795)
\end{itemize}
